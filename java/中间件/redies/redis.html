                                                                                    @Configuration
public class RedisConfig {
    //æœåŠ¡å™¨åœ°å€
    @Value("${spring.redis.host}")
    private String host;
    //æœåŠ¡å™¨ç«¯å£
    @Value("${spring.redis.port}")
    private int port;
    //è®¿é—®å¯†ç 
    @Value("${spring.redis.password}")
    private String password;
    //è¿æ¥è¶…æ—¶æ—¶é—´
    @Value("${spring.redis.timeout}")
    private String timeout;
    //æœ€å¤§è¿æ¥æ•°
    @Value("${spring.redis.jedis.pool.max-active}")
    private int maxTotal;
    //æœ€å¤§è¿æ¥é˜»å¡ç­‰å¾…æ—¶é—´
    @Value("${spring.redis.jedis.pool.max-wait}")
    private String maxWait;
    //æœ€å¤§ç©ºé—²è¿æ¥æ•°é‡
    @Value("${spring.redis.jedis.pool.max-idle}")
    private int maxIdle;
    //æœ€å°ç©ºé—²è¿æ¥æ•°é‡
    @Value("${spring.redis.jedis.pool.min-idle}")
    private int minIdle;
    @Bean
    public JedisPool getJedisPool(){
        JedisPoolConfig jedisPoolConfig=new JedisPoolConfig();
        jedisPoolConfig.setMaxTotal(maxTotal);
        jedisPoolConfig.setMaxWaitMillis(Long.valueOf(maxWait.substring(0,maxWait.length()-2)));
        jedisPoolConfig.setMaxIdle(maxIdle);
        jedisPoolConfig.setMinIdle(minIdle);
        JedisPool jedisPool=new JedisPool(jedisPoolConfig,host,port,Integer.valueOf(timeout.substring(0,timeout.length()-2)),password);
        return jedisPool;
    }
}

    @AfterAll
    public void closetConn(){
        if (jedis!=null){

        }
    }
